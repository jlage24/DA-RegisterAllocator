299template <class T>

309/*
310 * Removes a vertex with a given content (in) from a graph (this), and
311 * all outgoing and incoming edges.
312 * Returns true if successful, and false if such vertex does not exist.
313 */
314template <class T>
315bool Graph<T>::removeVertex(const T &in) {
316 for (auto it = vertexSet.begin(); it != vertexSet.end(); it++)
317 if ((*it)->info == in) {
318 auto v = *it;
319 vertexSet.erase(it);
320 for (auto u : vertexSet)
321 u->removeEdgeTo(v);
322 delete v;
323 return true;
324 }
325 return false;
326}
