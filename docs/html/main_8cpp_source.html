36static InterferenceGraph runAllocation(std::vector<Web> &webs,
37 const RegisterConfig &config) {
38 auto ig = InterferenceGraphBuilder::build(webs);
39 if (config.algorithm == "basic") {
40 RegisterAllocator::allocateBasic(ig, config.numRegisters);
41 } else if (config.algorithm == "spilling") {
42 SpillAllocator::allocate(ig, config.numRegisters, config.algorithmParam);
43 } else if (config.algorithm == "splitting") {
44 ig = SplitAllocator::allocate(webs, config.numRegisters, config.algorithmParam);
45 } else if (config.algorithm == "free") {
46 ig = FreeAllocator::allocate(webs, config.numRegisters);
47 } else {
48 throw std::runtime_error("Unknown algorithm: " + config.algorithm);
49 }
50 return ig;
51}
